# KK-17 自动化赏金猎人 Cron 配置
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin:/usr/local/sbin:/usr/sbin:/sbin
HOME=/root

# 加载环境变量
BASH_ENV=/root/.openclaw/workspace/projects/bounty-hunter-pro/.env

# 1. 每小时执行赏金猎人（使用启动脚本确保环境变量加载）
0 * * * * cd /root/.openclaw/workspace/projects/bounty-hunter-pro && bash automation/run_hunter.sh >> /root/.openclaw/workspace/projects/bounty-hunter-pro/automation/logs/cron.log 2>&1

# 2. 每天 09:00 执行赚钱机会搜索
0 9 * * * cd /root/.openclaw/workspace/projects/bounty-hunter-pro && bash automation/run_hunter.sh >> /root/.openclaw/workspace/projects/bounty-hunter-pro/automation/logs/cron.log 2>&1

# 3. 清理日志（每周一次）
0 0 * * 0 find /root/.openclaw/workspace/projects/bounty-hunter-pro/automation/logs -name "*.log" -mtime +7 -delete
